    <script>
        // Impede que o navegador restaure a posição de rolagem anterior
        history.scrollRestoration = 'manual';

        const splash = document.getElementById('splash');
        const mainContent = document.getElementById('main-content');

        // Função de scroll suave com offset (apenas mobile)
        function scrollToSectionMobile(sectionId) {
            const targetSection = document.getElementById(sectionId);
            if (!targetSection) return;
            const headerHeight = document.getElementById('siteHeader').offsetHeight;
            const sectionTop = targetSection.getBoundingClientRect().top + window.pageYOffset;
            window.scrollTo({
                top: sectionTop - headerHeight - 10,
                behavior: 'smooth'
            });
        }

        window.addEventListener('load', () => {
            setTimeout(() => {
                splash.style.transition = 'opacity 0.5s ease';
                splash.style.opacity = '0';

                setTimeout(() => {
                    splash.style.display = 'none';
                    mainContent.style.transition = 'opacity 0.7s ease';
                    mainContent.style.opacity = '1';

                    const hash = window.location.hash.substring(1);
                    const isMobile = window.innerWidth <= 992;

                    if (hash && hash !== 'home' && document.getElementById(hash)) {
                        if (isMobile) {
                            scrollToSectionMobile(hash);
                        }
                        // No desktop, o navegador trata a âncora normalmente
                    } else {
                        // Rolagem instantânea para o topo ao abrir sem hash
                        window.scrollTo(0, 0);
                    }

                    setTimeout(() => {
                        document.querySelectorAll('.hero .fade-in-up').forEach((el, i) => {
                            setTimeout(() => {
                                el.classList.add('appear');
                            }, i * 200);
                        });
                    }, 300);
                }, 700);
            }, 2000);
        });

        // Clique nos links do menu
        document.querySelectorAll('.nav-link').forEach(link => {
            link.addEventListener('click', function (e) {
                const targetId = this.getAttribute('href').substring(1);
                const isMobile = window.innerWidth <= 992;

                if (isMobile) {
                    e.preventDefault();
                    if (targetId === 'home') {
                        window.scrollTo(0, 0);
                    } else {
                        scrollToSectionMobile(targetId);
                    }
                }
                // No desktop, comportamento padrão

                // Fecha menu mobile se aberto
                hamburger.classList.remove('active');
                navMenu.classList.remove('active');
            });
        });

        // ✅ OBSERVER CORRIGIDO: anima apenas uma vez e nunca remove a classe "appear"
        const observer = new IntersectionObserver((entries) => {
            entries.forEach(entry => {
                if (entry.isIntersecting) {
                    entry.target.classList.add('appear');
                    // Para de observar o elemento após a animação, mantendo-o visível para sempre
                    observer.unobserve(entry.target);
                }
                // ❌ Removido o else que tirava a classe "appear"
            });
        }, { threshold: 0.15, rootMargin: '0px 0px -50px 0px' });

        document.querySelectorAll('.fade-in-up:not(.hero .fade-in-up)').forEach(el => observer.observe(el));

        const hamburger = document.getElementById('hamburger');
        const navMenu = document.getElementById('navMenu');

        hamburger.addEventListener('click', () => {
            hamburger.classList.toggle('active');
            navMenu.classList.toggle('active');
        });

        // Atualiza link ativo no scroll
        const sections = document.querySelectorAll('section[id]');
        window.addEventListener('scroll', () => {
            let current = '';
            sections.forEach(section => {
                const sectionTop = section.offsetTop - 100;
                if (window.scrollY >= sectionTop) {
                    current = section.getAttribute('id');
                }
            });
            document.querySelectorAll('.nav-link').forEach(link => {
                link.classList.remove('active');
                if (link.getAttribute('href') === '#' + current) {
                    link.classList.add('active');
                }
            });
        });
    </script>
